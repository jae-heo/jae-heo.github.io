Python asyncio 내부 구조와 커스텀 이벤트 루프 구현

이산 이벤트 시뮬레이션(DES) 엔진을 구축할 때, Python의 asyncio는 강력한 기반을 제공하지만 성능이 중요한 애플리케이션에서는 내부 구조를 이해하는 것이 필수적입니다.

이벤트 루프 아키텍처 이해하기

asyncio 이벤트 루프는 본질적으로 정교한 스케줄링 메커니즘을 갖춘 콜백의 우선순위 큐입니다:

import heapq
import collections
from dataclasses import dataclass, field

@dataclass(order=True)
class Event:
    time: float
    priority: int = field(compare=False)
    callback: callable = field(compare=False)
    args: tuple = field(default_factory=tuple, compare=False)

class SimulationEventLoop:
    def __init__(self):
        self._ready = collections.deque()
        self._scheduled = []
        self._current_time = 0.0
        self._event_count = 0

    def call_at(self, when: float, callback, *args, priority=0):
        """특정 시뮬레이션 시간에 이벤트 스케줄링"""
        event = Event(when, priority, callback, args)
        heapq.heappush(self._scheduled, event)
        self._event_count += 1
        return event

    def run_until(self, stop_time: float):
        """지정된 시뮬레이션 시간까지 실행"""
        while self._scheduled and self._scheduled[0].time <= stop_time:
            event = heapq.heappop(self._scheduled)
            self._current_time = event.time
            event.callback(*event.args)

Zero-Copy 이벤트 전파 구현

시뮬레이션 엔진에서 이벤트 전파 오버헤드는 런타임을 지배할 수 있습니다. 메모리 뷰와 객체 풀을 사용한 zero-copy 구현:

import numpy as np
from typing import Optional

class EventPool:
    """객체 재사용을 위한 이벤트 풀"""
    def __init__(self, size: int = 100_000):
        self._pool = []
        self._available = collections.deque()

        # 사전 할당된 numpy 배열로 메모리 레이아웃 최적화
        self._event_data = np.zeros(
            size,
            dtype=[
                ('time', 'f8'),
                ('entity_id', 'i8'),
                ('event_type', 'i4'),
                ('data', 'f8', (10,))
            ]
        )

        for i in range(size):
            self._available.append(i)

    def acquire(self) -> int:
        """이벤트 인덱스 획득"""
        if not self._available:
            raise MemoryError("Event pool exhausted")
        return self._available.popleft()

    def release(self, idx: int):
        """이벤트 인덱스 반환"""
        self._event_data[idx] = 0  # Clear data
        self._available.append(idx)

고급 스케줄링 알고리즘

Calendar Queue를 사용한 O(1) 이벤트 스케줄링:

class CalendarQueue:
    """O(1) 평균 시간 복잡도를 위한 Calendar Queue"""
    def __init__(self, bucket_width: float = 1.0, n_buckets: int = 256):
        self.bucket_width = bucket_width
        self.n_buckets = n_buckets
        self.buckets = [[] for _ in range(n_buckets)]
        self.n_events = 0
        self.last_bucket = 0
        self.bucket_top = 0.0

    def enqueue(self, time: float, event):
        """이벤트 삽입 - 평균 O(1)"""
        bucket_idx = int(time / self.bucket_width) % self.n_buckets

        # 버킷 내 정렬 삽입
        bucket = self.buckets[bucket_idx]
        insertion_point = 0
        for i, (t, _) in enumerate(bucket):
            if t > time:
                insertion_point = i
                break
            insertion_point = i + 1

        bucket.insert(insertion_point, (time, event))
        self.n_events += 1
