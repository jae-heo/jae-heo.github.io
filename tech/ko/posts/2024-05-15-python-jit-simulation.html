시뮬레이션을 위한 Python JIT 컴파일러 최적화

JIT(Just-In-Time) 컴파일러를 활용하면 Python 시뮬레이션 코드를 네이티브 성능에 근접하게 실행할 수 있습니다. 다양한 JIT 솔루션의 심화 활용법을 살펴봅니다.

Numba를 활용한 고성능 시뮬레이션

import numba as nb
from numba import cuda, prange, vectorize
import numpy as np
from numba.typed import Dict, List
from numba.experimental import jitclass

# JIT 클래스 정의
spec = [
    ('n_entities', nb.int64),
    ('states', nb.float64[:, :]),
    ('events', nb.float64[:]),
    ('time', nb.float64),
]

@jitclass(spec)
class NumbaSimulation:
    """Numba JIT 컴파일된 시뮬레이션 엔진"""

    def __init__(self, n_entities):
        self.n_entities = n_entities
        self.states = np.zeros((n_entities, 10), dtype=np.float64)
        self.events = np.zeros(n_entities, dtype=np.float64)
        self.time = 0.0

    @nb.njit(fastmath=True, parallel=True)
    def update_states(self, dt):
        """병렬화된 상태 업데이트"""
        for i in prange(self.n_entities):
            # SIMD 최적화 가능한 벡터 연산
            for j in range(10):
                self.states[i, j] += dt * self._compute_derivative(i, j)

    @nb.njit(inline='always')
    def _compute_derivative(self, entity_id, state_idx):
        """인라인 최적화된 미분 계산"""
        return np.exp(-self.states[entity_id, state_idx] * 0.1)

# GPU 가속 시뮬레이션
@cuda.jit
def cuda_simulation_kernel(states, events, dt, result):
    """CUDA 커널로 구현된 시뮬레이션"""
    # 쓰레드 인덱싱
    tx = cuda.threadIdx.x
    ty = cuda.threadIdx.y
    bx = cuda.blockIdx.x
    by = cuda.blockIdx.y
    bw = cuda.blockDim.x
    bh = cuda.blockDim.y

    # 글로벌 인덱스 계산
    i = tx + bx * bw
    j = ty + by * bh

    # 공유 메모리 활용
    shared_cache = cuda.shared.array(shape=(32, 32), dtype=nb.float64)

    if i < states.shape[0] and j < states.shape[1]:
        # 공유 메모리로 로드
        shared_cache[tx, ty] = states[i, j]
        cuda.syncthreads()

        # 계산 수행
        local_sum = 0.0
        for k in range(min(32, states.shape[1])):
            local_sum += shared_cache[tx, k] * dt

        # 결과 저장
        result[i, j] = local_sum

# 벡터화 함수
@vectorize([nb.float64(nb.float64, nb.float64)],
           target='parallel', fastmath=True)
def vectorized_interaction(x, y):
    """SIMD 벡터화된 상호작용 계산"""
    return np.exp(-abs(x - y)) / (1.0 + x * y)

JAX를 활용한 자동 미분과 XLA 컴파일

import jax
import jax.numpy as jnp
from jax import jit, grad, vmap, pmap
from functools import partial
import haiku as hk

class JAXSimulation:
    """JAX 기반 미분 가능한 시뮬레이션"""

    def __init__(self, n_entities: int):
        self.n_entities = n_entities

        # PRNGKey 초기화
        self.key = jax.random.PRNGKey(0)

    @partial(jit, static_argnums=(0,))
    def simulate_step(self, state, params, dt):
        """JIT 컴파일된 시뮬레이션 스텝"""
        # 자동 미분 가능한 연산
        def entity_dynamics(entity_state, entity_params):
            # Neural ODE 스타일 다이나믹스
            hidden = jnp.tanh(entity_state @ entity_params['W1'])
            output = hidden @ entity_params['W2']
            return output

        # vmap을 통한 벡터화
        batched_dynamics = vmap(entity_dynamics, in_axes=(0, None))

        # 상태 업데이트
        derivatives = batched_dynamics(state, params)
        new_state = state + dt * derivatives

        return new_state

    @partial(jit, static_argnums=(0,))
    def compute_loss(self, state, target, params):
        """손실 함수 (자동 미분용)"""
        predicted = self.simulate_step(state, params, 0.01)
        loss = jnp.mean((predicted - target) ** 2)
        return loss

    def optimize_parameters(self, initial_state, target_state, params):
        """경사하강법으로 파라미터 최적화"""
        # 그래디언트 함수 생성
        grad_fn = jit(grad(self.compute_loss, argnums=2))

        # Adam 옵티마이저
        import optax
        optimizer = optax.adam(learning_rate=1e-3)
        opt_state = optimizer.init(params)

        @jit
        def update_step(params, opt_state, state, target):
            grads = grad_fn(state, target, params)
            updates, opt_state = optimizer.update(grads, opt_state)
            params = optax.apply_updates(params, updates)
            return params, opt_state

        # 최적화 루프
        for _ in range(1000):
            params, opt_state = update_step(
                params, opt_state, initial_state, target_state
            )

        return params

# XLA 백엔드 최적화
@partial(jit, backend='gpu')  # GPU에서 XLA 컴파일
def xla_optimized_simulation(states, params):
    """XLA 최적화된 시뮬레이션"""
    # 퓨전 가능한 연산 체인
    x = jnp.dot(states, params['W1'])
    x = jax.nn.relu(x)
    x = jnp.dot(x, params['W2'])
    x = jax.nn.softmax(x, axis=-1)

    # 스캔을 통한 시퀀셜 처리
    def scan_fn(carry, x):
        state = carry
        state = state * 0.9 + x * 0.1
        return state, state

    _, outputs = jax.lax.scan(scan_fn, x[0], x)
    return outputs

# 다중 디바이스 병렬화
@pmap
def parallel_simulation_step(state, params):
    """다중 GPU/TPU 병렬 실행"""
    # 각 디바이스에서 독립적 실행
    local_result = xla_optimized_simulation(state, params)

    # 디바이스 간 통신 (all-reduce)
    global_mean = jax.lax.pmean(local_result, axis_name='devices')

    return global_mean

PyPy JIT 최적화 패턴

# PyPy에서 최적화되는 패턴들
class PyPyOptimizedSimulation:
    """PyPy JIT 친화적인 시뮬레이션"""

    def __init__(self, n_entities):
        self.n_entities = n_entities
        # PyPy는 리스트를 효율적으로 처리
        self.entities = [Entity(i) for i in range(n_entities)]

        # 타입 안정적인 딕셔너리
        self.event_handlers = {}
        self._register_handlers()

    def _register_handlers(self):
        """JIT 친화적인 핸들러 등록"""
        # 가상 함수 호출 피하기
        self.event_handlers[EVENT_MOVE] = self._handle_move
        self.event_handlers[EVENT_INTERACT] = self._handle_interact

    def process_events(self, events):
        """타입 안정적인 이벤트 처리"""
        # PyPy는 타입이 안정적인 루프를 최적화
        for event in events:
            handler = self.event_handlers.get(event.type)
            if handler is not None:
                handler(event)

    def _handle_move(self, event):
        """인라인 가능한 핸들러"""
        entity = self.entities[event.entity_id]
        # 간단한 연산은 인라인됨
        entity.x += event.dx
        entity.y += event.dy

    # 루프 언롤링을 위한 힌트
    def update_all_entities(self):
        """PyPy가 최적화하는 루프 패턴"""
        # 루프 분할로 최적화 유도
        n = self.n_entities

        # 4개씩 언롤링
        i = 0
        while i < n - 3:
            self.entities[i].update()
            self.entities[i+1].update()
            self.entities[i+2].update()
            self.entities[i+3].update()
            i += 4

        # 나머지 처리
        while i < n:
            self.entities[i].update()
            i += 1

하이브리드 JIT 전략

import numba as nb
import jax.numpy as jnp
from jax import jit

class HybridJITSimulation:
    """다중 JIT 컴파일러 활용"""

    def __init__(self):
        self.use_gpu = self._detect_gpu()

    @staticmethod
    @nb.njit(cache=True)
    def numba_critical_loop(data, n_iterations):
        """Numba로 컴파일된 크리티컬 루프"""
        result = np.zeros_like(data)

        for i in range(n_iterations):
            # CPU 집약적 계산
            for j in range(len(data)):
                result[j] += np.exp(-data[j] * i / 1000.0)

        return result

    @staticmethod
    @jit
    def jax_matrix_operations(matrices):
        """JAX로 최적화된 행렬 연산"""
        # GPU에서 효율적인 대규모 행렬 연산
        result = matrices[0]
        for m in matrices[1:]:
            result = jnp.matmul(result, m)
            result = jax.nn.relu(result)

        return result

    def run_hybrid_simulation(self, data, matrices):
        """하이브리드 실행"""
        # CPU 작업은 Numba로
        cpu_result = self.numba_critical_loop(data, 100)

        # GPU 작업은 JAX로
        if self.use_gpu:
            gpu_result = self.jax_matrix_operations(matrices)
        else:
            gpu_result = self._numpy_fallback(matrices)

        return cpu_result, gpu_result

커스텀 JIT 컴파일러 확장

from numba import types
from numba.core import cgutils
from numba.core.imputils import lower_builtin
from numba.core.typing import signature

# 커스텀 인트린식 정의
@nb.intrinsic
def custom_atomic_add(typingctx, val_t, ptr_t):
    """원자적 덧셈 연산"""
    sig = types.void(val_t, ptr_t)

    def codegen(context, builder, sig, args):
        val, ptr = args
        # LLVM IR 생성
        old = builder.atomic_rmw('add', ptr, val, 'seq_cst')
        return context.get_dummy_value()

    return sig, codegen

# 사용 예시
@nb.njit
def parallel_accumulate(array, indices, values):
    """원자적 누적"""
    for i in range(len(indices)):
        custom_atomic_add(values[i], array.ctypes.data + indices[i])
